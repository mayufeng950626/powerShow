查看jdk版本
java -version



安装mysql
yum install mysql
yum install mysql-server
yum install mysql-devel




service mysqld start      --启动mysql
service mysqld stop       --关闭mysql・
lsof -i:3306              --数据库端口是否开启


查看mysql版本
mysql --version




登录mysql
mysql -u root -p
